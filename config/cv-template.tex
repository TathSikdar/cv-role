% Bullet item. Two things here are load-bearing for ATS parsing, do not
% "tidy" them away:
%   1. No stray spaces or newlines inside the macro body. A space before the
%      \vspace lands in horizontal mode and can push an empty line box onto the
%      page, which shows up as a mystery gap between bullets.
%   2. \linespread tightens leading INSIDE a bullet so that the gap between a
%      bullet's wrapped lines stays smaller than the gap between two bullets.
%      pdftotext infers paragraph breaks from relative line spacing, so if that
%      ordering inverts, every wrapped bullet extracts as two separate
%      paragraphs and ATS parsers read one bullet as two records.
% scripts/measure_spacing.py verifies both properties.
\newcommand{\resumeItem}[1]{\item{\small\linespread{0.85}\selectfont #1\par}\vspace{0.5pt}}
\newcommand{\resumeSubheading}[4]{
  \vspace{-3pt}\item
    \begin{tabular*}{1.0\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & \textbf{\small #2} \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-7pt}
}
\newcommand{\resumeSubheadingContinue}[2]{
  \vspace{-3pt}
    \begin{tabular*}{1.0\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-7pt}
}
